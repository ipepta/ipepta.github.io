{
  "socketJson": "{
                  "socketId": "0b89aff1-68ed-4b5e-82e1-e9a5aea1aa97",
                  "messageId": "2f4983f6-4c7c-4238-864f-3f613b448134",
                  "version": 3389,
                  "command": 0,
                  "dataCount": 3,
                  "dataList":[
                               {
                                  "dataItemIndex": 0,
                                  "id": 7500,
                                  "value": 15.50,
                                  "timeStamp": "2019/02/20 22:03:59",
                                  "exception": null
                               },
                               ...
                             ],
                  "dataHex": "33464339344530312D4334363.....",

dataCount int 数据数量
dataList List 数据列表
dataItemIndex int 数据序号
id int 数据字段ID
value double 数据字段值
timeStamp string 数据时间
exception object socket消息错误或数据项错误
dataHex string 原始数据
createdDate datetime socket消息时间
listenerSignature string socket监听器签名
